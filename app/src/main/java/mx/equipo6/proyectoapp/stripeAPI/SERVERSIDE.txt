const express = require('express');
const bodyParser = require('body-parser');
const stripe = require('stripe')('sk_test_51Q5tAn1L09JDsSP2N2jjq4WuCu3m2JEfsYZsijP6mJmv3S185zPlH748NbWUMv6qAKBw5xV6kpGDztLFILLQafUJ00CC00kiUk');
const app = express();

app.use(bodyParser.json());

app.post('/create-payment-intent', async (req, res) => {
    const { amount, payment_method } = req.body;

    try {
        // Create a PaymentIntent using the PaymentMethod ID
        const paymentIntent = await stripe.paymentIntents.create({
            amount: amount,
            currency: 'usd',
            payment_method: payment_method,
            confirmation_method: 'automatic',
            confirm: true,
            return_url: 'https://github.com/XxCppSlayerxX/stripeTestSuccess/blob/main/index.html', // Replace with your return URL
            // Uncomment if you want to avoid redirect-based payment methods
            /*
            automatic_payment_methods: {
                enabled: true,
                allow_redirects: 'never'
            }
            */
        });

        res.json({ clientSecret: paymentIntent.client_secret });
    } catch (error) {
        console.error('Error creating payment intent:', error);
        res.status(500).send({ error: error.message });
    }
});

const PORT = 3000;
app.listen(PORT, () => {
    console.log(`Server is running on port ${PORT}`);
});
